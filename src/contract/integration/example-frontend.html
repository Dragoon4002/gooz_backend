üìù Code Example

// Frontend Code (React/Vue/Vanilla JS)

// 1. Connect Wallet
const provider = new ethers.providers.Web3Provider(window.ethereum);
await provider.send("eth_requestAccounts", []);
const signer = provider.getSigner();
const playerAddress = await signer.getAddress();

// 2. Request deposit from backend
const response = await fetch('/api/deposit/prepare', {
    method: 'POST',
    headers: { 'Content-Type': 'application/json' },
    body: JSON.stringify({
        gameId: 'game-123',
        playerAddress: playerAddress
    })
});

const { transaction } = await response.json();
// transaction = {to, data, value, chainId}

// 3. Player signs with MetaMask
const tx = await signer.sendTransaction(transaction);
console.log('Transaction sent:', tx.hash);

// 4. Wait for confirmation
await tx.wait();
console.log('Transaction confirmed!');

// 5. Notify backend
await fetch('/api/deposit/confirm', {
    method: 'POST',
    headers: { 'Content-Type': 'application/json' },
    body: JSON.stringify({
        gameId: 'game-123',
        playerAddress: playerAddress,
        txHash: tx.hash
    })
});
